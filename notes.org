## Possible McmcClasses

## Wide
## matrix with columns: parameter1, parameter2
setClass("McmcWide1", contains="data.table",
         representation(
             iteration="integer",
             chain="integer",
             parameters="character",
             template="list",
             indices="list"))
## Data.frame with columns
## chain, iteration, parameter1, parameter2, ...
setClass("McmcWide2", contains="data.frame",
         representation(parameters="character",
                        template="list",
                        indices="list"))
## Data.table with columns:
## chain, iteration, parameter1, parameter2, ...
setClass("McmcWide3", contains="data.table",
         representation(parameters="character",
                        template="list",
                        indices="list"))

## data.frame or data.table with columns
## parameter, chain, iteration, value
setClass("McmcLong", contains="data.frame",
         representation(parameters="character",
                        template="list",
                        indices="list"))

## List of matrices?
## Each element could be for a single parameter
## either long or wide.
setClass("ListOfMatrices", contains="list")
setClass("McmcList", contains="ListOfMatrices",
         representation(parameters="character",
                        template="list",
                        indices="list"))
